# stepsの外側にあるrunはID名の可能性
# run: dateはただただ現在時刻を出力するシェルコマンド

┌─ 分    (0-59)
│  ┌─ 時    (0-23)
│  │  ┌─ 日    (1-31)
│  │  │  ┌─ 月    (1-12)
│  │  │  │  ┌─ 曜日  (0-7, 0と7が日曜)
│  │  │  │  │
*  *  *  *  *

'0 9 * * 1-5'    # 平日(月〜金)の9:00に実行
'0 0 * * 0'      # 毎週日曜の0:00に実行
'30 18 1 * *'    # 毎月1日の18:30に実行
'*/15 * * * *'   # 15分ごとに実行

*/はステップ値。*/nごとという意味

#steps内の「-」はステップの開始を表す

steps:
  └─ [0] (ステップ1)
       └─ run: echo "${EXAMPLE}"

  └─ [1] (ステップ2)
       ├─ env:
       │    └─ EXAMPLE: Defined by step level
       └─ run: echo "${EXAMPLE}"
